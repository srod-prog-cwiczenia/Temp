\begin{definition}
  Niech dla $r\in [0;1]$:
  \[
    \zbiorPe{r} = \lbrace x \in \cantor \colon \lim_{n\to\infty}
    \frac{|\lbrace x = 1 \rbrace \cap [0; n)|}{n} = r\rbrace.
  \]
  \[
    \zbiorA{r} = \lbrace x \in \cantor \colon \lim_{k\to\infty}
    \frac{|\lbrace x = 1 \rbrace \cap [2^k; 2^{k+1})|}{2^k} = r\rbrace.
  \]
\end{definition}

\begin{theorem}
  $\zbiorPe{r} \subseteq \zbiorA{r}$
\end{theorem}
\begin{proof}
  Niech $X \subseteq \omega$.
  Wybierzmy $\eta > 0$.
  Wybieramy $K$ takie że dla $k \geq K$
  mamy
  \[
r - \eta \leq \frac{|X \cap [0; 2^k)|}{2^k} \leq r + \eta.
\]
Mamy zatem:
$|[2^k; 2^{k+1}) \cap X| = |[0; 2^{k+1}) \cap X| - |[0; 2^k) \cap X|$
$\leq (1 + \eta) \cdot 2^{k+1} - (r - \eta) \cdot 2^k$
$=2^k \cdot (r + 3\eta)$.
\Big(ponieważ:
$\frac{|X \cap [0; 2^k)|}{2^k} \geq r - \eta$, czyli
$- |X \cap [0; 2^k)| \leq -(r - \eta) 2^k$
\Big).

Jednocześnie mamy: 
$|[2^k; 2^{k+1}) \cap X| = |[0; 2^{k+1}) \cap X| - |[0; 2^k) \cap X|$
$\geq (1 - \eta) \cdot 2^{k+1} - (r + \eta) \cdot 2^k =$
$=2^k \cdot (r - 3\eta)$.
\Big(ponieważ:
$\frac{|X \cap [0; 2^k)|}{2^k} \leq r + \eta$, czyli
$- |X \cap [0; 2^k)| \geq -(r + \eta) 2^k$
\Big).

Zatem 
\[r - 3 \eta \leq \frac{|X \cap [2^k; 2^{k + 1})|}{2^k} \leq r + 3 \eta\]
(dla $k\geq K$).
Dowodzi to że $X \in \zbiorA{r}$.
\end{proof}
  
Zauważmy też że $\zbiorA{0} = \zbiorPe{0}$ oraz
$\zbiorA{1} = \zbiorPe{1}$.
